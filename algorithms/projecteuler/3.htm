from math import sqrt

def erat(n):
    # Create a list filled with ints from 2 to n, inclusive.
    numbers = range(2, n+1)

    # Remove all multiples of 2
    i = 0
    if numbers[i] == 2:
        i = 1

    length = len(numbers)
    limit = int(sqrt(n))

    j = 0
    while numbers[j] <= limit:
        i = 2
        prime_check = numbers[j]
        while i < length:
            if numbers[i] % prime_check == 0:
                numbers.pop(i)
                length -= 1
            else:
                i += 1
        j += 1

    return numbers

def solve():
    max = 600851475143
    primes = erat(10000)

    factor = 0
    for i in primes:
        if max % i == 0:
            factor = i

    return factor

if __name__ == "__main__":
    print solve()
